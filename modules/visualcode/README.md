# Install
```bash
brew cask install visual-studio-code

code --install-extension msjsdiag.debugger-for-chrome
code --install-extension esbenp.prettier-vscode
code --install-extension editorconfig.editorconfig
code --install-extension akamud.vscode-theme-onedark
code --install-extension flowtype.flow-for-vscode
code --install-extension ms-vscode.js-atom-grammar
code --install-extension robertohuertasm.vscode-icons
code --install-extension felixfbecker.php-intellisense
code --install-extension mikeburgh.xml-format
code --install-extension ms-vsliveshare.vsliveshare

code --disable-extension vscode.typescript-language-features

cp settings.json keybindings.json ~/Library/Application\ Support/Code/User/
```
